Cube Progress Indicator

A 3x3 grid progress indicator arranged like a cube's face

Usage

Multiple Modes on Same Page

<!-- Progress indicator -->
<div id="progress-indicator"></div>

<!-- Train animation -->
<div id="train-indicator"></div>

<!-- Another progress with different pattern -->
<div id="progress-2"></div>

<script>
// Progress mode - manual control
const progressIndicator = new CubeProgressIndicator('#progress-indicator', {
    pattern: 'sequential',
    mode: 'progress'
});
progressIndicator.setProgress(75);

// Train mode - auto-starts animation
const trainIndicator = new CubeProgressIndicator('#train-indicator', {
    pattern: 'anticlockwise-inward',
    mode: 'train',
    autoStart: true
});

// Another progress indicator with different pattern
const progress2 = new CubeProgressIndicator('#progress-2', {
    pattern: 'diagonal-lr',
    mode: 'progress'
});
progress2.setProgress(50);
</script>

Size Control

<div id="small-indicator"></div>
<div id="large-indicator"></div>

<script>
// Preset sizes: 'small', 'medium', 'large', 'xlarge'
const small = new CubeProgressIndicator('#small-indicator', {
    size: 'small'  // 15px container, 3px dots
});

const large = new CubeProgressIndicator('#large-indicator', {
    size: 'large'  // 30px container, 6px dots
});

// Custom size in pixels
const custom = new CubeProgressIndicator('#custom-indicator', {
    size: 50,           // 50px container
    dotSize: 8,         // 8px dots
    gap: 6              // 6px gap between dots
});

// Change size dynamically
custom.setSize('xlarge');              // Change to xlarge preset
custom.setSize(60, 10, 8);            // Custom: 60px container, 10px dots, 8px gap
</script>

Speed and Timing Control

<div id="speed-indicator"></div>

<script>
// Create indicator with custom speeds
const indicator = new CubeProgressIndicator('#speed-indicator', {
    pattern: 'anticlockwise-inward',
    mode: 'train',
    autoStart: true,
    trainSpeed: 100,        // Train animation speed in ms (lower = faster)
    trainPauseDelay: 300,  // Pause between cycles in ms
    animationSpeed: 200     // Progress animation speed in ms per dot
});

// Change speeds dynamically
indicator.setTrainSpeed(50);        // Make train faster (50ms)
indicator.setTrainPauseDelay(200); // Shorter pause (200ms)
indicator.setAnimationSpeed(150);   // Faster progress animation

// Get current speeds
const currentSpeed = indicator.getTrainSpeed();        // Returns: 50
const currentPause = indicator.getTrainPauseDelay();   // Returns: 200
const currentAnimSpeed = indicator.getAnimationSpeed(); // Returns: 150
</script>

Simple - Single Instance

<div class="cube-progress-indicator"></div>

<script>
// Progress mode (default)
const indicator = new CubeProgressIndicator('.cube-progress-indicator', {
    pattern: 'anticlockwise-inward',
    mode: 'progress',
    size: 'medium'  // Optional: 'small', 'medium', 'large', 'xlarge', or custom px
});
indicator.setProgress(75); // Set progress to 75%

// Train mode - auto-starts train animation
const trainIndicator = new CubeProgressIndicator('.train-indicator', {
    pattern: 'anticlockwise-inward',
    mode: 'train',
    autoStart: true,
    size: 'large'  // Larger size for train animation
});
</script>

Available Patterns

  • Sequential: Left-right, top-bottom (1â†’9)
  • Vertical: Top-bottom, left-right
  • Reverse: Right-left, bottom-top
  • Anticlockwise Inward: Spiral inward counter-clockwise
  • Clockwise Inward: Spiral inward clockwise
  • Anticlockwise Outward: Spiral outward counter-clockwise
  • Clockwise Outward: Spiral outward clockwise
  • Diagonal LR: Top-left to bottom-right diagonal
  • Diagonal RL: Top-right to bottom-left diagonal
  • Zigzag Horizontal: Horizontal zigzag pattern
  • Zigzag Vertical: Vertical zigzag pattern
  • Center Outward: Center to corners
  • Corners First: Corners then center
  • Snake: Snake-like pattern
  • Random: Random order
